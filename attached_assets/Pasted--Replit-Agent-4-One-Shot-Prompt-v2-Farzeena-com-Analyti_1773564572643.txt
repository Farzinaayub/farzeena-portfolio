## 🔐 AUTHENTICATION — BETTER-AUTH + MAGIC LINK + RESEND

This is the most critical part. Implement it **exactly** as described.

### How it works
1. Admin visits `/admin/login`, enters their email address
2. App calls `authClient.signIn.magicLink({ email, callbackURL: '/admin/dashboard' })`
3. better-auth calls the `sendMagicLink` function in `/lib/auth.ts`
4. `sendMagicLink` uses **Resend** to email a one-time magic link to the admin
5. Admin clicks the link in their email → better-auth verifies the token → session cookie set → redirect to `/admin/dashboard`
6. `disableSignUp: true` — only pre-existing users in the DB can log in (no self-registration)

### `/lib/auth.ts` — Server config
```typescript
import { betterAuth } from 'better-auth'
import { mongodbAdapter } from 'better-auth/adapters/mongodb'
import { magicLink } from 'better-auth/plugins'
import { Resend } from 'resend'
import { client } from './mongodb' // MongoClient singleton

const resend = new Resend(process.env.RESEND_API_KEY)

export const auth = betterAuth({
  database: mongodbAdapter(client, { dbName: 'farzeena-portfolio' }),
  plugins: [
    magicLink({
      disableSignUp: true,         // Only pre-seeded admins can log in
      expiresIn: 60 * 15,          // Link expires in 15 minutes
      sendMagicLink: async ({ email, url }) => {
        await resend.emails.send({
          from: 'Farzeena Admin <noreply@farzeena.com>',
          to: email,
          subject: 'Your Admin Login Link — farzeena.com',
          html: `
            <div style="font-family:Inter,sans-serif;max-width:480px;margin:auto;padding:32px">
              <h2 style="color:#0F172A">Sign in to Farzeena Admin</h2>
              <p style="color:#374151">Click the button below to sign in. This link expires in 15 minutes and can only be used once.</p>
              <a href="${url}" style="display:inline-block;margin-top:16px;padding:12px 24px;background:#1D4ED8;color:#fff;border-radius:8px;text-decoration:none;font-weight:600">
                Sign In to Admin
              </a>
              <p style="color:#6B7280;margin-top:24px;font-size:13px">If you didn't request this, ignore this email.</p>
            </div>
          `,
        })
      },
    }),
  ],
})
```

### `/middleware.ts` — Protect all admin routes
```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getSessionFromRequest } from 'better-auth/next-js'

export async function middleware(request: NextRequest) {
  const session = await getSessionFromRequest(request)
  const isAdminRoute = request.nextUrl.pathname.startsWith('/admin')
  const isLoginPage = request.nextUrl.pathname === '/admin/login'
  const isMagicLinkSentPage = request.nextUrl.pathname === '/admin/magic-link-sent'

  if (isAdminRoute && !isLoginPage && !isMagicLinkSentPage && !session) {
    return NextResponse.redirect(new URL('/admin/login', request.url))
  }
  return NextResponse.next()
}

export const config = {
  matcher: ['/admin/:path*'],
  runtime: 'nodejs',   // Required — avoid Edge runtime issues with MongoDB
}
```

## 🔧 UTILITY FUNCTIONS (`/lib/utils.ts`)

Implement these utilities — they are used throughout the app:

```typescript
// Convert Google Drive share link → direct embed URL
// Input:  https://drive.google.com/file/d/FILE_ID/view?usp=sharing
// Output: https://drive.google.com/uc?export=view&id=FILE_ID
export function getGoogleDriveImageUrl(shareUrl: string): string {
  const match = shareUrl.match(/\/d\/([\w-]+)/)
  if (!match) return shareUrl
  return `https://drive.google.com/uc?export=view&id=${match[1]}`
}

// Generate URL slug from title
export function generateSlug(title: string): string {
  return slugify(title, { lower: true, strict: true, trim: true })
}

// Estimate reading time from Markdown content
export function estimateReadingTime(content: string): number {
  const wordsPerMinute = 200
  const words = content.trim().split(/\s+/).length
  return Math.ceil(words / wordsPerMinute)
}
```
